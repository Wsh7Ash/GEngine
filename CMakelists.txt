function(ge_prefer_local_fetchcontent dep_name)
    if(NOT GE_USE_LOCAL_FETCHCONTENT_CACHE)
        return()
    endif()

    string(TOLOWER "${dep_name}" dep_lower)
    string(TOUPPER "${dep_name}" dep_upper)

    set(local_candidates
        "${CMAKE_SOURCE_DIR}/deps/${dep_lower}"
        "${CMAKE_SOURCE_DIR}/build/_deps/${dep_lower}-src"
    )

    foreach(candidate IN LISTS local_candidates)
        if(EXISTS "${candidate}")
            set("FETCHCONTENT_SOURCE_DIR_${dep_upper}" "${candidate}" PARENT_SCOPE)
            set("FETCHCONTENT_UPDATES_DISCONNECTED_${dep_upper}" ON PARENT_SCOPE)
            message(STATUS "Using local ${dep_name} source from ${candidate}")
            return()
        endif()
    endforeach()
endfunction()
